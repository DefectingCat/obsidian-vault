## 按钮

![](Pasted%20image%2020240520141042.png)

新版热点的按钮中，`spot` 字段就是 `SpotContent`

```ts
/**
 * 用户自定义按钮
 */
export interface CustomButton extends IconData, NameData {
    id: string;
    /** 按钮位置 */
    type: ButtonPositions;
    /** 序号 */
    index: number;
    /**
     * 按钮显示内容，同热点内容
     */
    spot: SpotContent;
    /**
     * 自定义样式
     */
    style?: CustomStyleBase;
    /**
     * 二级菜单
     */
    children?: CustomButton[] | null;
}
```
